Query For Redo Log Information 


select   a.MEMBER,

         b.GROUP#,

         b.THREAD#,

         b.SEQUENCE#,

         b.BYTES,

         b.MEMBERS,

         b.ARCHIVED,

         b.STATUS,

         b.FIRST_CHANGE#,

         b.FIRST_TIME

from     sys.v_$logfile a, sys.v_$log b

where    a.GROUP# = b.GROUP#

order by a.MEMBER;
