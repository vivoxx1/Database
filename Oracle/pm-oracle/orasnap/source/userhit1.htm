Query For User Hit Ratio 


select   USERNAME,

         CONSISTENT_GETS,

         BLOCK_GETS,

         PHYSICAL_READS,

         ((CONSISTENT_GETS+BLOCK_GETS-PHYSICAL_READS) / (CONSISTENT_GETS+BLOCK_GETS))

from     sys.v_$session a, sys.v_$sess_io b

where    a.SID = b.SID

and      (CONSISTENT_GETS+BLOCK_GETS) > 0

and      USERNAME is not null

order by ((CONSISTENT_GETS+BLOCK_GETS-PHYSICAL_READS) / (CONSISTENT_GETS+BLOCK_GETS));
