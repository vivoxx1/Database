Query For Open Cursors By User 


select 

  thread#,

  sequence#,

  name,

  first_change#,

  TO_CHAR(first_time, 'mm/dd/yyyy HH24:MI:SS'),

  next_change#,

  TO_CHAR(next_time, 'mm/dd/yyyy HH24:MI:SS'),

  (blocks * block_size),

   archived,

   applied,

   deleted,

   status

FROM

    v$archived_log

WHERE

    status in ('A')

ORDER BY

    thread#

  , sequence#;
