Query For Average Length Write Queue Length 


select   sum(decode(NAME,'summed dirty queue length',VALUE)),

         sum(decode(NAME,'write requests',VALUE)),

         sum(decode(NAME,'summed dirty queue length',VALUE)) / 

         sum(decode(NAME,'write requests',VALUE))

from     sys.v_$sysstat

where    NAME in ('summed dirty queue length','write requests')

and      VALUE > 0;
