Query For Session Stats (By Session) 


select   nvl(ss.USERNAME,'ORACLE PROC'),

         se.SID,

         sn.NAME

         VALUE

from     sys.v_$session ss, sys.v_$sesstat se, sys.v_$statname sn

where    se.STATISTIC# = sn.STATISTIC#

and      se.SID = ss.SID

and      se.VALUE > 0

order by sn.NAME, se.SID, se.VALUE desc;
