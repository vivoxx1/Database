Query For Session I/O By User 


select   nvl(ses.USERNAME,'ORACLE PROC'),

         OSUSER,

         PROCESS,

         ses.SID,

         SERIAL#,

         PHYSICAL_READS,

         BLOCK_GETS,

         CONSISTENT_GETS,

         BLOCK_CHANGES,

         CONSISTENT_CHANGES

from     sys.v_$session ses, sys.v_$sess_io sio

where    ses.SID = sio.SID

order by PHYSICAL_READS, ses.USERNAME;
