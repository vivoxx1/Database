Query For Index Level Partitioning Information 


select   INDEX_OWNER,

         INDEX_NAME,

         PARTITION_NAME,

         HIGH_VALUE_LENGTH,

         PARTITION_POSITION,

         STATUS,

         TABLESPACE_NAME,

         PCT_FREE,

         INI_TRANS,

         MAX_TRANS,

         INITIAL_EXTENT,

         NEXT_EXTENT,

         MIN_EXTENT,

         MAX_EXTENT,

         PCT_INCREASE,

         FREELISTS,

         FREELIST_GROUPS,

         LOGGING,

         BLEVEL,

         LEAF_BLOCKS,

         DISTINCT_KEYS,

         AVG_LEAF_BLOCKS_PER_KEY,

         AVG_DATA_BLOCKS_PER_KEY,

         CLUSTERING_FACTOR,

         NUM_ROWS,

         SAMPLE_SIZE,

         LAST_ANALYZED,

         BUFFER_POOL

from     dba_ind_partitions

where    INDEX_OWNER not in ('SYS','SYSTEM')

order by INDEX_OWNER, INDEX_NAME, PARTITION_NAME;
