Query For Shared Pool Details 


select   OWNER,

         NAME,

         DB_LINK,

         NAMESPACE,

         TYPE,

         SHARABLE_MEM,

         LOADS,

         EXECUTIONS,

         LOCKS,

         PINS,

         KEPT

from     sys.v_$db_object_cache

where    rownum < 1001           -- limit number of rows returned

order by OWNER, NAME;
