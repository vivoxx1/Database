Query For Database Snapshots 


select   OWNER,

         NAME,

         TABLE_NAME,

         MASTER_VIEW,

         MASTER_OWNER,

         MASTER,

         MASTER_LINK,

         CAN_USE_LOG,

         UPDATABLE,

         LAST_REFRESH,

         ERROR,

         TYPE,

         NEXT,

         REFRESH_GROUP

from     dba_snapshots

order by OWNER, NAME;
