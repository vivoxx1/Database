Query For Database Clusters 


select   a.OWNER,

         TABLESPACE_NAME,

         a.CLUSTER_NAME,

         TABLE_NAME,

         TAB_COLUMN_NAME,

         CLU_COLUMN_NAME

from     dba_clusters a, dba_clu_columns b

where    a.CLUSTER_NAME = b.CLUSTER_NAME

and      a.OWNER not in ('SYS', 'SYSTEM')

order by a.OWNER, TABLESPACE_NAME, a.CLUSTER_NAME, TABLE_NAME;
