Query For Large Objects 


select   owner,

         segment_name,

         segment_type,

         tablespace_name,

         bytes,

         initial_extent,

         next_extent,

         extents,

         pct_increase

from 	 dba_segments

where    bytes > (

         select avg(bytes) 

         from   dba_segments

         where  owner not in ('SYS','SYSTEM'))

and      owner  not in ('SYS','SYSTEM')

order by bytes desc;
