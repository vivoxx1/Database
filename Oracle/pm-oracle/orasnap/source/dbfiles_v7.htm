Query For Database File Information 


select   'Archived Log Directory',

         value

from     sys.v_$parameter

where    name = 'log_archive_dest'

UNION ALL

select   'Control Files',

         value

from     sys.v_$parameter

where    name = 'control_files'

UNION ALL

select   'Datafile',

         name

from     sys.v_$datafile

UNION ALL

select   'LogFile Member',

         member

from     sys.v_$logfile;
