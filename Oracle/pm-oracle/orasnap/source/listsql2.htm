Query For Shared Pool With Most Buffer Scans 


select   EXECUTIONS,

         BUFFER_GETS,

         round(DISK_READS / decode(EXECUTIONS, 0, 1, EXECUTIONS) / 400,2),

         SQL_TEXT

from     sys.v_$sqlarea

where    BUFFER_GETS / decode(EXECUTIONS,0,1, EXECUTIONS) / 400 > 10 

order by EXECUTIONS desc;
