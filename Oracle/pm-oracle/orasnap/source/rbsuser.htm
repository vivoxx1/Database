Query For Rollback Users 


select   a.NAME,

         b.XACTS,

         c.SID,

         c.SERIAL#,

         c.USERNAME,

         c.OSUSER,

         d.SQL_TEXT

from     sys.v_$rollname a, sys.v_$rollstat b, sys.v_$session c, 

         sys.v_$sqlarea d, sys.v_$transaction e

where    a.USN = b.USN

and      b.USN = e.XIDUSN

and      c.TADDR = e.ADDR

and      c.SQL_ADDRESS = d.ADDRESS

and      c.SQL_HASH_VALUE = d.HASH_VALUE

order by a.NAME, c.SID;
