Query For Two Phase Commit Pending Information 


select   LOCAL_TRAN_ID,

         GLOBAL_TRAN_ID,

         STATE,

         MIXED,

         ADVICE,

         TRAN_COMMENT,

         FAIL_TIME,

         FORCE_TIME,

         RETRY_TIME,

         OS_USER,

         OS_TERMINAL,

         HOST,

         DB_USER,

         COMMIT#

from     dba_2pc_pending

order by LOCAL_TRAN_ID, GLOBAL_TRAN_ID;
