Query For Tablespace Coalesced Extents 


select   TABLESPACE_NAME,

         TOTAL_EXTENTS,

         EXTENTS_COALESCED,

         PERCENT_EXTENTS_COALESCED,

         TOTAL_BYTES,

         BYTES_COALESCED,

         TOTAL_BLOCKS,

         BLOCKS_COALESCED,

         PERCENT_BLOCKS_COALESCED

from     dba_free_space_coalesced

order by TABLESPACE_NAME;
