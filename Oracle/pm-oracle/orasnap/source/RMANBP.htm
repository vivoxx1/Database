Query For Open Cursors By User 


SELECT  

  bs.recid,

  bp.piece#,

  bp.copy# ,

  Status,

  Handle,

  TO_CHAR(bp.start_time, 'mm/dd/yyyy HH24:MI:SS'),

  TO_CHAR(bp.completion_time, 'mm/dd/yyyy HH24:MI:SS'),

  bp.elapsed_seconds

FROM

   v$backup_set    bs

  ,v$backup_piece  bp

WHERE

      bs.set_stamp = bp.set_stamp

  AND bs.set_count = bp.set_count

  AND bp.status IN ('A', 'X')

ORDER BY

    bs.recid

  , piece#;
