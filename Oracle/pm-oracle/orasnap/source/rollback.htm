Query For Rollback Segment Information 


select   SEGMENT_NAME,

         OWNER,

         TABLESPACE_NAME,

         SEGMENT_ID,

         FILE_ID,

         BLOCK_ID,

         INITIAL_EXTENT,

         NEXT_EXTENT,

         MIN_EXTENTS,

         MAX_EXTENTS,

         PCT_INCREASE,

         STATUS,

         INSTANCE_NUM

from     dba_rollback_segs

order by SEGMENT_NAME;
