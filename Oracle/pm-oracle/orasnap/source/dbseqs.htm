Query For Database Sequences 


select   SEQUENCE_OWNER,

         SEQUENCE_NAME,

         MIN_VALUE,

         MAX_VALUE,

         INCREMENT_BY,

         CYCLE_FLAG,

         ORDER_FLAG,

         CACHE_SIZE,

         LAST_NUMBER

from     dba_sequences

where    SEQUENCE_OWNER not in ('SYS','SYSTEM')

order by SEQUENCE_OWNER,SEQUENCE_NAME;
